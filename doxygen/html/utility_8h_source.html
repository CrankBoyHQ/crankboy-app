146// compute the next highest power of 2 of 32-bit v,
147// or v if v is a power of 2
148// https://stackoverflow.com/a/466242
149static inline unsigned next_pow2(unsigned v)
150{
151 --v;
152 v |= v >> 1;
153 v |= v >> 2;
154 v |= v >> 4;
155 v |= v >> 8;
156 v |= v >> 16;
157 return ++v;
158}
